\tikzset{
  % MMD输出节点样式
  dan loss output/.style={
    rounded corners=2pt, % 轻微圆角
    draw=gray!65, % 边框颜色
    fill=gray!8, % 填充颜色
    minimum width=15mm, % 最小宽度
    minimum height=8mm, % 最小高度
    align=center, % 居中换行
    font=\scriptsize % 使用较小字号
  },
  % DAN-MMD共享CFE组件
  pics/dan mmd shared cfe/.style={
    code={ % 定义DAN-MMD共享特征提取器
      % scale控制CFE细节模块在整张DAN-MMD图里的显示大小
      \begin{scope}[scale=.48, transform shape] % 缩小CFE细节模块
        \pic {cfe module}; % 引用CFE模块
      \end{scope}
      \node[fit=(cfebox), inner sep=0pt] (danCFE) {}; % 共享CFE区域
      % 下面四个坐标是外部箭头连接点，都落在CFE外框边界上
      \coordinate (dan-cfe-source-west) at ([yshift=4mm]danCFE.west); % 源域输入接外框
      \coordinate (dan-cfe-target-west) at ([yshift=-4mm]danCFE.west); % 目标域输入接外框
      \coordinate (dan-src-feature-east) at ([yshift=4mm]danCFE.east); % 源域特征从外框输出
      \coordinate (dan-tgt-feature-east) at ([yshift=-4mm]danCFE.east); % 目标域特征从外框输出
    }
  },
  % MMD损失组件
  pics/dan mmd loss/.style={
    code={ % 定义MMD损失模块
      % scale控制两个特征点云的大小，位置由下方布局参数决定
      \begin{scope}[scale=.62, transform shape] % 缩小域特征点云
        \pic at (0, 0) {domain feature pair}; % MMD特征分布
      \end{scope}
      \node[dan loss output] (lossOut) at (1.85, 0) {MMD\\Loss}; % 损失输出
      \draw[dann confuse edge] (domainFeature-east) -- (lossOut.west); % MMD到loss
      \coordinate (mmd-west) at (-.40, 0); % 外部CFE到MMD的入口点
    }
  }
}
